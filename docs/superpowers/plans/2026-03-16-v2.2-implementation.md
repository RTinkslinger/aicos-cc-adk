# Architecture v2.2 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the v1 three-agent architecture with v2.2: State MCP (5 tools for CAI), Web Tools MCP (restructured, async task pattern), Content Agent (Bash+skills, CC-like), Sync Agent (background Notion sync).

**Architecture:** Four processes on droplet. Two MCPs (State, Web Tools) serve CAI via Cloudflare tunnels. Two agents (Content, Sync) run internally with scheduled sessions, using Bash+psql+skills like CC. Communication via Postgres (inbox, notifications). See `docs/superpowers/specs/2026-03-15-architecture-v2.2-design.md`.

**Tech Stack:** claude-agent-sdk 0.1.48, FastMCP 3.x (streamable-http), asyncpg, Playwright, Python 3.11+, systemd, Cloudflare Tunnel.

**Codebase:** `mcp-servers/agents/` (Mac) → `/opt/agents/` (droplet via deploy.sh)

asyncpg connection pool with lifespan pattern for FastMCP.

```python
import asyncpg
import os

_pool: asyncpg.Pool | None = None

async def get_pool() -> asyncpg.Pool:
    global _pool
    if _pool is None:
        _pool = await asyncpg.create_pool(os.environ["DATABASE_URL"], min_size=2, max_size=5)
    return _pool

async def close_pool():
    global _pool
    if _pool:
        await _pool.close()
        _pool = None
```

```python
import asyncio
import uuid
from dataclasses import dataclass, field
from datetime import datetime

@dataclass
class WebTask:
    task_id: str
    task: str
    url: str
    status: str = "started"  # started, running, complete, error
    result: dict | None = None
    error: str | None = None
    started_at: datetime = field(default_factory=datetime.utcnow)
    completed_at: datetime | None = None
    cost_usd: float | None = None

_tasks: dict[str, WebTask] = {}

def create_task(task: str, url: str = "") -> WebTask:
    t = WebTask(task_id=f"wt_{uuid.uuid4().hex[:12]}", task=task, url=url)
    _tasks[t.task_id] = t
    return t

def get_task(task_id: str) -> WebTask | None:
    return _tasks.get(task_id)

def complete_task(task_id: str, result: dict, cost_usd: float = 0):
    t = _tasks.get(task_id)
    if t:
        t.status = "complete"
        t.result = result
        t.cost_usd = cost_usd
        t.completed_at = datetime.utcnow()

def fail_task(task_id: str, error: str):
    t = _tasks.get(task_id)
    if t:
        t.status = "error"
        t.error = error
        t.completed_at = datetime.utcnow()
```

```python
"""Content Agent runner — asyncio timers trigger ClaudeSDKClient sessions."""
import asyncio
import os
from pathlib import Path
from claude_agent_sdk import query, ClaudeAgentOptions, AgentDefinition, ResultMessage

INBOX_INTERVAL = 60      # 1 minute
CONTENT_INTERVAL = 300   # 5 minutes

def load_system_prompt() -> str:
    return Path(__file__).parent.joinpath("system_prompt.md").read_text()

def get_options() -> ClaudeAgentOptions:
    return ClaudeAgentOptions(
        model="claude-sonnet-4-6",
        system_prompt=load_system_prompt(),
        permission_mode="dontAsk",
        allowed_tools=[
            "Bash", "Read", "Write", "Grep", "Glob",
            "Agent", "Skill",
            "mcp__web__web_browse", "mcp__web__web_scrape",
            "mcp__web__web_search", "mcp__web__extract_youtube",
            "mcp__web__extract_transcript", "mcp__web__fingerprint",
            "mcp__web__check_strategy", "mcp__web__manage_session",
            "mcp__web__validate", "mcp__web__cookie_status",
            "mcp__web__watch_url",
        ],
        mcp_servers={
            "web": {"type": "http", "url": "http://localhost:8001/mcp"},
        },
        setting_sources=["project"],
        thinking={"type": "enabled", "budget_tokens": 10000},
        effort="high",
        cwd="/opt/agents",
        env={
            "ANTHROPIC_API_KEY": os.environ.get("ANTHROPIC_API_KEY", ""),
            "DATABASE_URL": os.environ.get("DATABASE_URL", ""),
            "FIRECRAWL_API_KEY": os.environ.get("FIRECRAWL_API_KEY", ""),
        },
    )

async def run_session(prompt: str):
    """Run a single agent session with the given prompt."""
    options = get_options()
    async for message in query(prompt=prompt, options=options):
        if isinstance(message, ResultMessage):
            print(f"Session complete: {message.subtype}, cost=${message.cost_usd:.4f}")

async def inbox_loop():
    """Check CAI inbox every 1 minute."""
    while True:
        try:
            await run_session(
                "Check the cai_inbox table for unprocessed messages. "
                "Process each one according to your inbox-handling skill. "
                "Mark processed when done."
            )
        except Exception as e:
            print(f"Inbox check error: {e}")
        await asyncio.sleep(INBOX_INTERVAL)

async def content_loop():
    """Run content pipeline every 5 minutes."""
    while True:
        try:
            await run_session(
                "Run your content pipeline cycle. Check all active sources "
                "in your watch list. Fetch new content. Analyze, score, "
                "publish digests, and write results to Postgres."
            )
        except Exception as e:
            print(f"Content cycle error: {e}")
        await asyncio.sleep(CONTENT_INTERVAL)

async def main():
    print("Content Agent starting...")
    await asyncio.gather(inbox_loop(), content_loop())

if __name__ == "__main__":
    asyncio.run(main())
```
